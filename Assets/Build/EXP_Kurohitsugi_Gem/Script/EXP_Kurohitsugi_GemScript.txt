local function SetActive(trans, active)
    if trans ~= nil then
        trans.gameObject:SetActive(active)
    end
end

local function SetChildrenActive(root, active)
    if root == nil then return end

    for i = 0, root.childCount - 1 do
        local child = root:GetChild(i)
        if child ~= nil then
            child.gameObject:SetActive(active)
        end
    end
end

local function PlayParticleSystemsInRoot(root)
    if root == nil then return end

    pcall(function()
        local systems = root:GetComponentsInChildren(typeof(UE.ParticleSystem), true)

        if systems ~= nil then
            for i = 0, systems.Length - 1 do
                local ps = systems[i]

                if ps ~= nil then
                    ps:Clear(true)
                    ps:Play(true)
                end
            end
        end
    end)
end

local function StopParticleSystemsInRoot(root)
    if root == nil then return end

    pcall(function()
        local systems = root:GetComponentsInChildren(typeof(UE.ParticleSystem), true)

        if systems ~= nil then
            for i = 0, systems.Length - 1 do
                local ps = systems[i]

                if ps ~= nil then
                    ps:Stop(true)
                    ps:Clear(true)
                end
            end
        end
    end)
end

local function SafeSetRendererMaterialFloat(renderer, propertyName, value)
    if renderer == nil then return end
    if propertyName == nil then return end

    -- Prima proviamo material: in Unity crea/usa l'istanza runtime del materiale del renderer.
    -- Così evitiamo di modificare direttamente l'asset materiale condiviso.
    pcall(function()
        if renderer.material ~= nil then
            renderer.material:SetFloat(propertyName, value)
        end
    end)

    -- Fallback su sharedMaterial: utile se, in BT/Lua, renderer.material non viene esposto.
    pcall(function()
        if renderer.sharedMaterial ~= nil then
            renderer.sharedMaterial:SetFloat(propertyName, value)
        end
    end)
end

local function Clamp01(v)
    if v < 0 then return 0 end
    if v > 1 then return 1 end
    return v
end

local function EaseOutCubic(t)
    t = Clamp01(t)
    return 1 - UE.Mathf.Pow(1 - t, 3)
end

local function EaseInCubic(t)
    t = Clamp01(t)
    return t * t * t
end

local function Lerp(a, b, t)
    return a + (b - a) * t
end

local function Vec3Lerp(a, b, t)
    return UE.Vector3(
        Lerp(a.x, b.x, t),
        Lerp(a.y, b.y, t),
        Lerp(a.z, b.z, t)
    )
end

local function IsVerticalLine(trans)
    if trans == nil then return false end

    local n = string.lower(trans.name)
    return string.find(n, "corner") ~= nil
end

local function GetHiddenTopLineScale(finalScale)
    if finalScale.x >= finalScale.z then
        return UE.Vector3(0.01, finalScale.y, finalScale.z)
    else
        return UE.Vector3(finalScale.x, finalScale.y, 0.01)
    end
end

local function GetHiddenSpikeScale(finalScale)
    return UE.Vector3(
        UE.Mathf.Max(finalScale.x * 0.20, 0.01),
        UE.Mathf.Max(finalScale.y * 0.03, 0.01),
        UE.Mathf.Max(finalScale.z * 0.20, 0.01)
    )
end

function spellFireBall:CalculateSequenceTimes()
    self.linesStartTime = self.delayBeforeStart
    self.topLinesStartTime = self.linesStartTime + self.verticalLinesDuration

    self.wallsStartTime = self.topLinesStartTime + self.topLinesDuration + self.linesDelay
    self.topStartTime = self.wallsStartTime + self.wallsGrowDuration + self.wallsDelay

    self.effectsTime = self.topStartTime + self.topGrowDuration + self.topDelay

    -- Nuova timeline:
    -- 1) il cubo/top si chiude;
    -- 2) parte la prima scarica;
    -- 3) poco dopo parte il lampo/glow con la seconda scarica;
    -- 4) le spike/fiamme si muovono in questa finestra e poi arriva la frantumazione.
    local topCompleteTime = self.topStartTime + self.topGrowDuration
    self.firstRainTime = topCompleteTime + self.firstRainAfterTopComplete
    self.secondRainTime = self.firstRainTime + self.secondRainAfterFirstRain

    self.glowTransitionTime = self.secondRainTime
    self.spikesTime = self.secondRainTime

    self.spikeRetractTime = self.spikesTime + self.spikeGrowDuration + self.holdAfterSpikes
    self.breakPiecesTime = self.spikeRetractTime + self.spikeRetractDuration

    self.hideTime = self.breakPiecesTime + self.breakPiecesDuration + self.breakPiecesHoldDuration

    print("KUROHITSUGI TIMING - topComplete=" .. tostring(topCompleteTime)
        .. " firstRain=" .. tostring(self.firstRainTime)
        .. " secondRain/glow=" .. tostring(self.secondRainTime)
        .. " glowEnd=" .. tostring(self.glowTransitionTime + self.glowTransitionDuration)
        .. " break=" .. tostring(self.breakPiecesTime))
end

                    line.localScale = UE.Vector3(0.06, 6, 0.06)
                elseif n == "Line_Corner_FR" then
                    line.localPosition = UE.Vector3(1.5, 3, 1.5)
                    line.localScale = UE.Vector3(0.06, 6, 0.06)
                elseif n == "Line_Corner_BL" then
                    line.localPosition = UE.Vector3(-1.5, 3, -1.5)
                    line.localScale = UE.Vector3(0.06, 6, 0.06)
                elseif n == "Line_Corner_BR" then
                    line.localPosition = UE.Vector3(1.5, 3, -1.5)
                    line.localScale = UE.Vector3(0.06, 6, 0.06)
                elseif n == "Line_Top_Front" then
                    line.localPosition = UE.Vector3(0, 6, 1.5)
                    line.localScale = UE.Vector3(3, 0.06, 0.06)
                elseif n == "Line_Top_Back" then
                    line.localPosition = UE.Vector3(0, 6, -1.5)
                    line.localScale = UE.Vector3(3, 0.06, 0.06)
                elseif n == "Line_Top_Left" then
                    line.localPosition = UE.Vector3(-1.5, 6, 0)
                    line.localScale = UE.Vector3(3, 0.06, 0.06)
                elseif n == "Line_Top_Right" then
                    line.localPosition = UE.Vector3(1.5, 6, 0)
                    line.localScale = UE.Vector3(3, 0.06, 0.06)
                end
            end
        end
    end

    if self.wallsRoot ~= nil then
        for i = 0, self.wallsRoot.childCount - 1 do
            local wall = self.wallsRoot:GetChild(i)
            if wall ~= nil then
                local n = wall.name

                if n == "Wall_Front" then
                    wall.localPosition = UE.Vector3(0, 3, 1.48)
                    wall.localRotation = UE.Quaternion.Euler(0, 0, 0)
                    wall.localScale = UE.Vector3(3, 6, 0.04)
                elseif n == "Wall_Back" then
                    wall.localPosition = UE.Vector3(0, 3, -1.48)
                    wall.localRotation = UE.Quaternion.Euler(0, 0, 0)
                    wall.localScale = UE.Vector3(3, 6, 0.04)
                elseif n == "Wall_Left" then
                    wall.localPosition = UE.Vector3(-1.48, 3, 0)
                    wall.localRotation = UE.Quaternion.Euler(0, 90, 0)
                    wall.localScale = UE.Vector3(3, 6, 0.04)
                elseif n == "Wall_Right" then

                    wall.localRotation = UE.Quaternion.Euler(0, 90, 0)
                    wall.localScale = UE.Vector3(3, 6, 0.04)

            end
        end
    end

    if self.topRoot ~= nil then
        for i = 0, self.topRoot.childCount - 1 do
            local top = self.topRoot:GetChild(i)
            if top ~= nil and top.name == "Top_Panel" then

function spellFireBall:GetFlatAimDirection()
    local dir = UE.Vector3(-self.trans.up.x, 0, -self.trans.up.z)

    if dir.sqrMagnitude < 0.01 then
        dir = UE.Vector3(0, 0, 1)
    end

    return dir.normalized
end

function spellFireBall:ClampFlatOffset(offset)
    local flat = UE.Vector3(offset.x, 0, offset.z)

    if flat.magnitude > self.maxTargetMove then
        flat = flat.normalized * self.maxTargetMove
    end

    return flat
end

function spellFireBall:GetRaycastArrayCount(hits)
    if hits == nil then return 0 end

    pcall(function()

            count = hits.Count
        end
    end)

    if count == nil then

    end

    return count
end

function spellFireBall:IsValidGroundCollider(col)
    if col == nil then return false end

    local valid = true

    -- Non appoggiamo mai il target su trigger/zone invisibili.
    pcall(function()
        if col.isTrigger then
            valid = false
        end
    end)

    -- Evita di colpire parti della gemma, preview, Kurohitsugi stesso, mani/player o oggetti tenuti.
    pcall(function()
        if valid and self:NameLooksLikePlayerOrHeldObject(col) then
            valid = false
        end
    end)

    pcall(function()
        if valid and col.transform ~= nil and self:IsTransformLinkedToPlayer(col.transform) then
            valid = false
        end
    end)

    pcall(function()
        if valid and self.trans ~= nil and col.transform ~= nil and col.transform.IsChildOf ~= nil and col.transform:IsChildOf(self.trans) then
            valid = false
        end
    end)

    return valid
end

function spellFireBall:IsValidGroundHit(point, normal, col)
    if point == nil then return false end
    if point.y < self.groundRayMinValidY then return false end
    if not self:IsValidGroundCollider(col) then return false end

    -- Evita che una parete, un muro o un fianco di collider venga considerato terreno.
    -- Un pavimento piano ha normal.y circa 1; una parete verticale ha normal.y circa 0.
    if normal ~= nil then
        if normal.y < self.groundMinNormalY then
            return false
        end
    end

    return true
end

function spellFireBall:GetHighestGroundPointAt(samplePos)
    if samplePos == nil then return nil end

    local origin = UE.Vector3(samplePos.x, samplePos.y + self.groundRayStartHeight, samplePos.z)
    local dir = UE.Vector3.down
    local bestPoint = nil
    local bestY = self.groundRayMinValidY

    local hits = nil

    pcall(function()
        hits = UE.Physics.RaycastAll(origin, dir, self.groundRayDistance)
    end)

    local count = self:GetRaycastArrayCount(hits)

    for i = 0, count - 1 do
        local hit = self:GetRaycastArrayItem(hits, i)
        local point = self:GetRaycastHitPoint(hit)
        local normal = self:GetRaycastHitNormal(hit)
        local col = self:GetRaycastHitCollider(hit)

        if self:IsValidGroundHit(point, normal, col) then
            -- Per ogni singolo raggio prendiamo la superficie valida più alta.
            -- Così non agganciamo pavimenti sotto la mappa o livelli inferiori attraversati dal raycast.
            if point.y > bestY then
                bestY = point.y
                bestPoint = point
            end
        end
    end

    return bestPoint
end

function spellFireBall:GetGroundSampleOffsets()
    local r = self.groundFootprintRadius
    local d = r * 0.72

    -- Campioniamo centro, assi e diagonali.
    -- Il cubo è largo circa 3 unità, quindi un raggio 1.35 copre quasi tutta la base senza uscire troppo.
    return {
        UE.Vector3(0, 0, 0),
        UE.Vector3(r, 0, 0),
        UE.Vector3(-r, 0, 0),
        UE.Vector3(0, 0, r),
        UE.Vector3(0, 0, -r),
        UE.Vector3(d, 0, d),
        UE.Vector3(-d, 0, d),
        UE.Vector3(d, 0, -d),
        UE.Vector3(-d, 0, -d)
    }
end

function spellFireBall:GetGroundedPosition(basePos, extraOffset)
    if basePos == nil then return basePos end

    if extraOffset == nil then
        extraOffset = 0
    end

    local fallback = UE.Vector3(basePos.x, basePos.y + extraOffset, basePos.z)
    local offsets = self:GetGroundSampleOffsets()
    local chosenY = nil

    for i = 1, #offsets do
        local o = offsets[i]
        local samplePos = UE.Vector3(basePos.x + o.x, basePos.y, basePos.z + o.z)
        local p = self:GetHighestGroundPointAt(samplePos)

        if p ~= nil then
            -- Tra le superfici reali sotto l'impronta del Kurohitsugi scegliamo la più bassa.
            -- Questo risolve le scale: il fondo del cubo parte dal gradino più basso coperto dalla base,
            -- evitando il vuoto visibile sotto una parte della struttura.
            -- Risolve anche il caso vicino ai muri: se il centro becca una parte alta del muro,
            -- i raggi laterali trovano il pavimento e vincono perché sono più bassi.
            if chosenY == nil or p.y < chosenY then
                chosenY = p.y
            end
        end
    end

    if chosenY ~= nil then
        return UE.Vector3(basePos.x, chosenY + extraOffset, basePos.z)
    end

    -- Fallback: se il raycast non trova nulla, non rompiamo la spell.
    -- Manteniamo la posizione calcolata prima.
    return fallback
end

function spellFireBall:InitTargetPosition()
    local flatDir = self:GetFlatAimDirection()

    self.startHandLocalPos = self:GetHandRelativePos()
    self.startTargetPos = self.trans.position + flatDir * self.targetDistance
    self.startTargetPos = self:GetGroundedPosition(self.startTargetPos, self.targetHeightOffset)
    self.lastTargetPos = self.startTargetPos

    if self.targetAreaRoot ~= nil then
        self.targetAreaRoot.position = self.startTargetPos

        local angleY = UE.Mathf.Atan2(flatDir.x, flatDir.z) * UE.Mathf.Rad2Deg
        self.targetAreaRoot.rotation = UE.Quaternion.Euler(0, angleY, 0)
    end
end

function spellFireBall:UpdateKurohitsugiPreviewManual()
    self:FindKurohitsugiObjects()

    self:UpdateChargeReiatsuBind()
    self:UpdateReiatsuVisual()

    if self.previewCube ~= nil then
        self.previewCube:Rotate(UE.Vector3(0.8, 1.4, 0.45) * UE.Time.deltaTime * self.rotationSpeed)
    end

    if self.targetAreaRoot ~= nil and self.startHandLocalPos ~= nil and self.startTargetPos ~= nil then
        local currentHandLocalPos = self:GetHandRelativePos()
        local handDelta = currentHandLocalPos - self.startHandLocalPos
        local targetOffset = self:ClampFlatOffset(handDelta * self.targetMoveMultiplier)

        local pos = self.startTargetPos + targetOffset
        pos = self:GetGroundedPosition(pos, self.targetHeightOffset)

        self.targetAreaRoot.position = pos

        local flatDir = self:GetFlatAimDirection()
        local angleY = UE.Mathf.Atan2(flatDir.x, flatDir.z) * UE.Mathf.Rad2Deg
        self.targetAreaRoot.rotation = UE.Quaternion.Euler(0, angleY, 0)

        self.lastTargetPos = pos
    end
end

function spellFireBall:AnimateLines()
    if self.lineData == nil then return end

    local verticalT = EaseOutCubic((self.sequenceTimer - self.linesStartTime) / self.verticalLinesDuration)
    local topT = EaseOutCubic((self.sequenceTimer - self.topLinesStartTime) / self.topLinesDuration)

    -- Dissolve delle linee: segue la loro materializzazione.
    -- Usiamo il valore più avanzato tra verticali e top linee, così il materiale diventa pieno
    -- man mano che l'intero telaio del Kurohitsugi viene disegnato.
    local lineDissolveT = verticalT
    if topT > lineDissolveT then
        lineDissolveT = topT
    end

    local lineDissolveAmount = Lerp(self.lineDissolveHiddenAmount, self.lineDissolveVisibleAmount, lineDissolveT)
    self:SetLineDissolveAmount(lineDissolveAmount)

    for i = 1, #self.lineData do
        local d = self.lineData[i]
        local tr = d.trans

        if d.vertical then
            local h = Lerp(0.01, d.scale.y, verticalT)
            tr.localScale = UE.Vector3(d.scale.x, h, d.scale.z)
            tr.localPosition = UE.Vector3(d.pos.x, h * 0.5, d.pos.z)
        else
            local startScale = GetHiddenTopLineScale(d.scale)
            tr.localScale = Vec3Lerp(startScale, d.scale, topT)
            tr.localPosition = d.pos
        end
    end
end

function spellFireBall:AnimateWalls()
    if self.wallData == nil then return end

    -- Crescita fisica delle pareti.
    local wallT = EaseOutCubic((self.sequenceTimer - self.wallsStartTime) / self.wallsGrowDuration)

    -- Reveal:
    -- il dissolve arriva a 1.20 esattamente quando le pareti finiscono di formarsi.
    -- Non usiamo wallDissolveExtraDuration qui, perché vogliamo il cambio subito dopo la crescita.
    local revealDuration = self.wallsGrowDuration
    local revealT = EaseOutCubic((self.sequenceTimer - self.wallsStartTime) / revealDuration)

    local revealAmount = Lerp(
        self.wallDissolveHiddenAmount,
        self.wallDissolveRevealAmount,
        revealT
    )

    -- Settle:
    -- appena le pareti hanno finito di salire, i materiali passano ai valori finali separati.
    local settleStartTime = self.wallsStartTime + self.wallsGrowDuration
    local settleT = Clamp01((self.sequenceTimer - settleStartTime) / self.wallDissolveSettleDuration)
    settleT = EaseOutCubic(settleT)

    if settleT <= 0 then
        self:SetWallDissolveAmount(revealAmount)
    else
        if self.wallDissolveMainRenderers == nil
            or self.wallDissolveLowerRenderers == nil
            or self.wallDissolveRenderers == nil
            or #self.wallDissolveRenderers == 0 then

            self:CacheWallDissolveRenderers()
        end

        local mainAmount = Lerp(
            self.wallDissolveRevealAmount,
            self.wallDissolveMainFinalAmount,
            settleT
        )

        local lowerAmount = Lerp(
            self.wallDissolveRevealAmount,
            self.wallDissolveLowerFinalAmount,
            settleT
        )

        self:SetWallRendererListDissolveAmount(self.wallDissolveMainRenderers, mainAmount)
        self:SetWallRendererListDissolveAmount(self.wallDissolveLowerRenderers, lowerAmount)
    end

    for i = 1, #self.wallData do
        local d = self.wallData[i]
        local tr = d.trans

        local h = Lerp(0.01, d.scale.y, wallT)
        tr.localScale = UE.Vector3(d.scale.x, h, d.scale.z)
        tr.localPosition = UE.Vector3(d.pos.x, h * 0.5, d.pos.z)
    end
end

function spellFireBall:AnimateTop()
    if self.topData == nil then return end

    local t = EaseOutCubic((self.sequenceTimer - self.topStartTime) / self.topGrowDuration)

    -- Preparazione futura: se assegneremo al top un materiale con _DissolveAmount,
    -- verrà animato automaticamente insieme alla chiusura del pannello.
    local dissolveAmount = Lerp(self.topDissolveHiddenAmount, self.topDissolveVisibleAmount, t)
    self:SetTopDissolveAmount(dissolveAmount)

    for i = 1, #self.topData do
        local d = self.topData[i]
        local tr = d.trans

        local startScale = UE.Vector3(d.scale.x * 0.05, d.scale.y, d.scale.z * 0.05)
        tr.localScale = Vec3Lerp(startScale, d.scale, t)
        tr.localPosition = d.pos
    end
end

function spellFireBall:GetBreakOutwardDirection(pos)
    local dir = UE.Vector3(pos.x, pos.y - 3.0, pos.z)

    if dir.sqrMagnitude < 0.001 then
        dir = UE.Vector3(0, 1, 0)
    end

    return dir.normalized
end

function spellFireBall:AnimateBreakPieces()
    if self.breakPieceData == nil then return end

    local t = EaseOutCubic((self.sequenceTimer - self.breakPiecesTime) / self.breakPiecesDuration)

    for i = 1, #self.breakPieceData do
        local d = self.breakPieceData[i]
        local tr = d.trans

        local outwardDir = self:GetBreakOutwardDirection(d.pos)
        local endPos = d.pos + outwardDir * self.breakPiecesOutwardDistance

        local startScale = d.scale * self.breakPiecesStartScaleMultiplier
        local endScale = d.scale * self.breakPiecesEndScaleMultiplier

        tr.gameObject:SetActive(true)
        tr.localPosition = Vec3Lerp(d.pos, endPos, t)
        tr.localScale = Vec3Lerp(startScale, endScale, t)

        tr:Rotate(
            UE.Vector3(outwardDir.x * 40, 35, outwardDir.z * 40) * UE.Time.deltaTime
        )
    end
end

function spellFireBall:IsTransformLinkedToPlayer(tr)
    if tr == nil then return false end

    local playerTr = self:GetPlayerTrans()
    if playerTr == nil then return false end

    pcall(function()
        if tr == playerTr then
            linked = true
        end
    end)

    pcall(function()
        if not linked and tr.IsChildOf ~= nil and tr:IsChildOf(playerTr) then
            linked = true
        end
    end)

    return linked
end

function spellFireBall:IsTooCloseToPlayer(pu)
    local playerTr = self:GetPlayerTrans()
    if playerTr == nil then return false end

    local pos = self:GetPhysicsUnitPosition(pu)
    if pos == nil then return false end

    local delta = pos - playerTr.position
    delta.y = 0

    return delta.magnitude <= self.gravityMinDistanceFromPlayer
end

function spellFireBall:IsValidGravityTarget(pu)
    if pu == nil then return false end

    local valid = true

    -- Evita scena/statica.
    pcall(function()
        if pu.IsStaticScene ~= nil and pu:IsStaticScene() then
            valid = false
        end
    end)

    pcall(function()
        if pu.IsScene ~= nil and pu:IsScene() then
            valid = false
        end
    end)

    -- Evita player diretto.
    pcall(function()
        local player = CL.InteractCharacter.PlayerInstance
        if player ~= nil and pu == player then
            valid = false
        end
    end)

    -- Evita la gemma stessa e componenti legati all'interact della gemma.
    pcall(function()
        if self.interact ~= nil and pu == self.interact then
            valid = false
        end
    end)

    -- Evita mani, braccia, gemma, armi, oggetti tenuti e qualunque transform figlia del player rig.
    local tr = self:GetPhysicsUnitTransform(pu)

    if self:IsTransformLinkedToPlayer(tr) then
        valid = false
    end

    if self:NameLooksLikePlayerOrHeldObject(pu) then
        valid = false
    end

    -- Filtro di sicurezza extra: se qualcosa è praticamente addosso al player, non lo tocchiamo.
    -- Serve a prevenire il bug delle braccia allungate in caso la query ritorni mani/oggetti del rig VR.
    if self:IsTooCloseToPlayer(pu) then
        valid = false
    end

    return valid
end

function spellFireBall:IsChargeReiatsuLikelyDead(pu)

    if not self.chargeReiatsuSkipLikelyDead then return false end

    local dead = false

    -- Nomi/proprietà diverse tra role, controller e ragdoll: proviamo più strade in pcall.
    pcall(function()
        if pu.isDead ~= nil and pu.isDead == true then
            dead = true
        end
    end)

    pcall(function()
        if not dead and pu.dead ~= nil and pu.dead == true then
            dead = true
        end
    end)

    pcall(function()
        if not dead and pu.IsDead ~= nil and pu:IsDead() then
            dead = true

    pcall(function()

        if not dead and pu.motion ~= nil and pu.motion.IsDead ~= nil and pu.motion:IsDead() then

    -- lo trattiamo come ragdoll/cadavere e non gli applichiamo più Reiatsu.
    pcall(function()
        if not dead and pu.actor == nil and pu.motion == nil and pu.controller == nil and pu.rb ~= nil then
            dead = true
        end
    end)

    return dead
end


function spellFireBall:FindChildTransformByNameRecursive(root, wantedNames)
    if root == nil then return nil end
    if wantedNames == nil then return nil end

    local rootName = ""

    pcall(function()
        if root.name ~= nil then
            rootName = string.lower(tostring(root.name))
        end
    end)

    for i = 1, #wantedNames do
        local wanted = wantedNames[i]
        if wanted ~= nil and rootName == string.lower(wanted) then
            return root
        end
    end

    local childCount = 0

    pcall(function()
        childCount = root.childCount
    end)

    if childCount == nil then childCount = 0 end

    for i = 0, childCount - 1 do
        local child = nil

        pcall(function()
            child = root:GetChild(i)
        end)

        local found = self:FindChildTransformByNameRecursive(child, wantedNames)
        if found ~= nil then
            return found
        end
    end

    return nil
end

function spellFireBall:GetChargeReiatsuChestTransform(target)
    if target == nil then return nil end

    local rootTr = self:GetPhysicsUnitTransform(target)
    if rootTr == nil then return nil end

    -- V3.7: priorità al bacino.
    -- Teniamo il nome della funzione per non toccare il resto dello script,
    -- ma ora il punto cercato è Hips/Pelvis prima di Chest/Spine.
    local exact = self:FindChildTransformByNameRecursive(rootTr, {
        "Hips",
        "Pelvis",
        "Bip001 Pelvis",
        "Bip001 Hips",
        "mixamorig:Hips",
        "mixamorig:Pelvis",
        "Spine",
        "Chest",
        "UpperChest"
    })

    if exact ~= nil then
        pcall(function()
            if self.chargeReiatsuPressureDebug then
                print("KUROHITSUGI HIPS DEBUG - exact pressure transform found=" .. tostring(exact.name))
            end
        end)
        return exact
    end

    -- Fallback elastico: alcuni prefab possono avere nomi composti tipo "GoblinSmall_Hips".
    local found = nil

    local function scanContains(tr)
        if tr == nil or found ~= nil then return end

        local n = ""
        pcall(function()
            if tr.name ~= nil then
                n = string.lower(tostring(tr.name))
            end
        end)

        if string.find(n, "hips") ~= nil or string.find(n, "pelvis") ~= nil then
            found = tr
            return
        end

        -- Fallback secondario: se proprio non trova bacino, prova spine/chest.
        if string.find(n, "spine") ~= nil or string.find(n, "chest") ~= nil then
            found = tr
            return
        end

        local count = 0
        pcall(function()
            count = tr.childCount
        end)
        if count == nil then count = 0 end

        for i = 0, count - 1 do
            local child = nil
            pcall(function()
                child = tr:GetChild(i)
            end)
            scanContains(child)
            if found ~= nil then return end
        end
    end

    scanContains(rootTr)

    pcall(function()
        if self.chargeReiatsuPressureDebug then
            if found ~= nil then
                print("KUROHITSUGI HIPS DEBUG - contains pressure transform found=" .. tostring(found.name))
            else
                print("KUROHITSUGI HIPS DEBUG - no Hips/Pelvis/Spine/Chest transform found on target=" .. tostring(self:GetSafeName(target)))
            end
        end
    end)

    return found
end

function spellFireBall:ApplyChargeReiatsuChestPressure(target, force)
    if target == nil then return false end
    if force == nil then return false end
    if not self.chargeReiatsuUseChestPressure then return false end

    local chestTr = self:GetChargeReiatsuChestTransform(target)
    if chestTr == nil then return false end

    local rb = self:GetRigidbodyFromTransform(chestTr)
    if rb == nil then
        pcall(function()
            if self.chargeReiatsuPressureDebug then
                print("KUROHITSUGI HIPS DEBUG - pressure transform found but rb nil transform=" .. tostring(chestTr.name))
            end
        end)
        return false
    end

    pcall(function()
        if self.chargeReiatsuPressureDebug then
            print("KUROHITSUGI HIPS DEBUG - pressure rb applied transform=" .. tostring(chestTr.name) .. " force=" .. tostring(force))
        end
    end)

    local applied = false
    local mul = self.chargeReiatsuChestForceMultiplier or 1.0

    pcall(function()
        rb.isKinematic = false
        rb:AddForce(force * mul, UE.ForceMode.Impulse)
        applied = true
    end)

    pcall(function()
        local v = rb.velocity
        if v ~= nil then
            local maxDown = self.chargeReiatsuMaxDownVelocity or -1.35
            local newY = v.y

            if newY < maxDown then
                newY = maxDown
            end

            local damping = self.chargeReiatsuChestVelocityDamping or self.chargeReiatsuVelocityDamping or 0.55

            rb.velocity = UE.Vector3(
                v.x * damping,
                newY,
                v.z * damping
            )
        end
    end)

    return applied
end

function spellFireBall:ApplyChargeReiatsuToPhysicsUnit(pu, center, doInterrupt)
    if pu == nil then return end
    if center == nil then return end

    -- V3.6: stessa filosofia della V3 originale, ma senza slam continuo.
    -- Risaliamo al Role Target come per la shockwave, ma NON usiamo IsValidGravityTarget().
    local target = pu

    if self.chargeReiatsuForceRoleTarget then
        local roleTarget = self:GetSpikeShockWaveRoleTarget(pu)
        if roleTarget ~= nil then
            target = roleTarget
        end
    end

    if target == nil then return end

    -- Evita player/gemma/mani, ma lascia passare i nemici validi.
    if not self:IsSpikeShockWaveValidTarget(target) then return end

    -- Nuovo filtro: non riattivare cadaveri/ragdoll già morti.
    if self:IsChargeReiatsuLikelyDead(target) then return end

    local puPos = self:GetPhysicsUnitPosition(target)
    if puPos == nil then return end

    -- V3.6: pressione fisica a micro-pulse ravvicinati.
    -- Non è forza continua ogni tick: è un impulso piccolo ma frequente,
    -- così il tremore sembra costante senza accumulare caduta infinita.
    local downForce = self.chargeReiatsuSmallEnemyDownForce or 4
    local force = UE.Vector3.down * downForce

    -- 1) Reazione muscolare: blocca postura/attacchi.
    pcall(function()
        if self.chargeReiatsuUseStiff and target.muscleState ~= nil then
            target.muscleState:DoStiff(self.chargeReiatsuStiffDuration)
        end
    end)

    -- 2) Dizzy: è la parte che hai confermato funzionare bene.
    pcall(function()
        if self.chargeReiatsuUseDizzy and target.muscleState ~= nil then
            target.muscleState:EnterDizzy(self.chargeReiatsuDizzyDuration)
        end
    end)

    -- 3) Stop dell'azione corrente: impedisce ai nemici di proseguire attacchi/movimenti.
    pcall(function()
        if self.chargeReiatsuStopAction and target.actor ~= nil then
            target.actor:StopAction(false)
        end
    end)

    -- 4) Motion protect/interruption breve, rinnovata a tick.
    pcall(function()
        if self.chargeReiatsuProtectMotion and target.motion ~= nil then
            target.motion:ProtectMotionFunctionForAWhile(self.chargeReiatsuProtectTime)
        end
    end)

    -- 5) Pulse throwing come V3, ma solo sui target vivi/validi.
    if doInterrupt and self.chargeReiatsuUseThrowingPulse then
        pcall(function()
            if target.muscleState ~= nil then
                target.muscleState:DoThrowing()
            end
        end)
    end

    -- 6) Knockout opzionale: resta spento di default.
    if self.chargeReiatsuKnockout and doInterrupt then
        pcall(function()
            if target.motion ~= nil then
                target.motion:Knockout()
            end
        end)
    end

    -- 7) Pressione fisica: non lasciamo i corpi in assenza di gravità troppo a lungo.
    pcall(function()
        if self.chargeReiatsuGravityOffTime ~= nil and self.chargeReiatsuGravityOffTime > 0 then
            target:DisableGravityAWhile(self.chargeReiatsuGravityOffTime)
        end
    end)

    pcall(function()
        target:SetKinematic(false)
    end)

    -- Impulso verso il basso SOLO a pulse, non ogni tick.
    -- doInterrupt è già scandito da chargeReiatsuKnockoutInterval, quindi ora la pressione fisica
    -- parte circa ogni 0.30s invece che 20 volte al secondo.
    if doInterrupt and self.chargeReiatsuSlamSmallEnemies then
        local chestApplied = self:ApplyChargeReiatsuChestPressure(target, force)

        -- Fallback: se non troviamo Hips/Pelvis/Spine/Chest, usiamo il target normale.
        -- Non applichiamo entrambe le forze, altrimenti torniamo al problema dello sprofondamento.
        if not chestApplied then
            pcall(function()
                target:AddForce(force, UE.ForceMode.Impulse)
            end)
        end
    end

    local rb = self:GetPhysicsUnitRigidbody(target)

    pcall(function()
        if rb ~= nil then
            -- Nessun rb:AddForce continuo: solo damping per frenare avanzamento e oscillazioni.
            local v = rb.velocity
            if v ~= nil then
                local maxDown = self.chargeReiatsuMaxDownVelocity or -1.35
                local newY = v.y

                -- Limite anti-sprofondamento: non lasciamo accumulare caduta artificiale.
                if newY < maxDown then
                    newY = maxDown
                end

                rb.velocity = UE.Vector3(
                    v.x * self.chargeReiatsuVelocityDamping,
                    newY,
                    v.z * self.chargeReiatsuVelocityDamping
                )
            end

            if rb.angularVelocity ~= nil then
                rb.angularVelocity = rb.angularVelocity * self.chargeReiatsuAngularDamping
            end
        end
    end)

    -- 8) Root fallback: manteniamo gli effetti di controllo, ma niente doppio slam forte sul root.
    pcall(function()
        if target.GetRootPu ~= nil then
            local root = target:GetRootPu()

            if root ~= nil and root ~= target and self:IsSpikeShockWaveValidTarget(root) and not self:IsChargeReiatsuLikelyDead(root) then
                if self.chargeReiatsuUseStiff and root.muscleState ~= nil then
                    root.muscleState:DoStiff(self.chargeReiatsuStiffDuration)
                end

                if self.chargeReiatsuUseDizzy and root.muscleState ~= nil then
                    root.muscleState:EnterDizzy(self.chargeReiatsuDizzyDuration)
                end

                if self.chargeReiatsuStopAction and root.actor ~= nil then
                    root.actor:StopAction(false)
                end

                if self.chargeReiatsuProtectMotion and root.motion ~= nil then
                    root.motion:ProtectMotionFunctionForAWhile(self.chargeReiatsuProtectTime)
                end

                if doInterrupt and self.chargeReiatsuUseRootInterrupt and self.chargeReiatsuUseThrowingPulse and root.muscleState ~= nil then
                    root.muscleState:DoThrowing()
                end

                local rootRb = self:GetPhysicsUnitRigidbody(root)
                local rootForceMul = self.chargeReiatsuRootForceMultiplier or 0.0

                if rootRb ~= nil then
                    if rootForceMul > 0 then
                        rootRb:AddForce(force * rootForceMul)
                    end

                    local rv = rootRb.velocity
                    if rv ~= nil then
                        local maxDown = self.chargeReiatsuMaxDownVelocity or -1.35
                        local rootY = rv.y * self.chargeReiatsuVelocityDamping

                        if rootY < maxDown then
                            rootY = maxDown
                        end

                        rootRb.velocity = UE.Vector3(
                            rv.x * self.chargeReiatsuVelocityDamping,
                            rootY,
                            rv.z * self.chargeReiatsuVelocityDamping
                        )
                    end

                    if rootRb.angularVelocity ~= nil then
                        rootRb.angularVelocity = rootRb.angularVelocity * self.chargeReiatsuAngularDamping
                    end
                end
            end
        end
    end)
end

function spellFireBall:ApplyChargeReiatsuTick()
    if not self.chargeReiatsuEnabled then return end
    if not self.chargeReiatsuActive then return end

    local center = self:GetChargeReiatsuCenter()
    local targets = nil

    -- Importante: solo logic target.
    -- Non usiamo QueryInRadiusTemp qui, così evitiamo di attirare archi/armi/oggetti secondari.
    pcall(function()
        targets = CL.PhysicsUnit.QueryLogicTargetInRadiusTemp(center, self.chargeReiatsuRadius)
    end)

    local count = self:GetListCount(targets)

    local doInterrupt = false
    local now = UE.Time.time

    if now - self.chargeReiatsuLastKnockoutTime >= self.chargeReiatsuKnockoutInterval then
        self.chargeReiatsuLastKnockoutTime = now
        doInterrupt = true
    end

    for i = 0, count - 1 do
        local pu = self:GetListItem(targets, i)
        self:ApplyChargeReiatsuToPhysicsUnit(pu, center, doInterrupt)
    end
end

function spellFireBall:UpdateChargeReiatsuBind()
    if not self.chargeReiatsuEnabled then return end
    if not self.chargeReiatsuActive then return end

    local now = UE.Time.time

    if now - self.chargeReiatsuLastTick >= self.chargeReiatsuTickInterval then
        self.chargeReiatsuLastTick = now
        self:ApplyChargeReiatsuTick()
    end
end


function spellFireBall:GetGravityRampMultiplier()
    if not self.gravityUseRamp then return 1.0 end
    if self.wallsGrowDuration == nil or self.wallsGrowDuration <= 0 then return 1.0 end

    -- Progressione delle pareti da 0 a 1.
    local wallProgress = (self.sequenceTimer - self.wallsStartTime) / self.wallsGrowDuration

    -- Vogliamo arrivare al massimo quando le pareti sono circa a metà.
    local maxAt = self.gravityRampMaxAtWallsProgress
    if maxAt == nil or maxAt <= 0.01 then
        maxAt = 0.50
    end

    local t = Clamp01(wallProgress / maxAt)
    t = EaseOutCubic(t)

    local startMul = self.gravityRampStartMultiplier
    if startMul == nil then
        startMul = 0.20
    end

    return Lerp(startMul, 1.0, t)
end

function spellFireBall:ApplyForceToPhysicsUnit(pu, force)
    if pu == nil then return end
    if force == nil then return end

    -- API Battle Talent / PhysicsUnit.
    pcall(function()
        pu:AddForce(force)
    end)

    -- Fallback diretto sul Rigidbody, utile quando QueryInRadiusTemp ritorna parti fisiche/ragdoll.
    pcall(function()
        if pu.rb ~= nil then
            pu.rb:AddForce(force)
        end
    end)

    -- Fallback ancora più aggressivo: piccola velocità diretta verso il centro.
    -- È volutamente moderata, per non ricreare il bug delle braccia.
    pcall(function()
        if pu.rb ~= nil then
            pu.rb.velocity = pu.rb.velocity + force.normalized * self.gravityVelocityAssist
        end
    end)
end

function spellFireBall:ApplyGravityToPhysicsUnit(pu, center)
    if pu == nil then return end
    if center == nil then return end
    if not self:IsValidGravityTarget(pu) then return end

    local puPos = self:GetPhysicsUnitPosition(pu)
    if puPos == nil then return end

    -- V3: il risucchio deve vincere l'ancoraggio a terra.
    -- Quindi tiriamo verso il centro alto + un piccolo lift, senza schiacciare verso il basso durante la fase di attrazione.
    local toCenter = center - puPos

    local pullDir = UE.Vector3(0, 0, 0)

    if toCenter.sqrMagnitude > 0.001 then
        pullDir = toCenter.normalized
    end

    local rampMul = self:GetGravityRampMultiplier()
    local force = pullDir * (self.gravityPullForce * rampMul)
        + UE.Vector3.up * (self.gravityLiftForce * rampMul)
        + UE.Vector3.down * (self.gravityDownForce * rampMul)

    -- Comportamento stile Black Hole: gravità spenta abbastanza a lungo e knockout.
    pcall(function()
        pu:DisableGravityAWhile(self.gravityOffTime)
    end)

    pcall(function()
        pu:SetKinematic(false)
    end)

    self:ApplyForceToPhysicsUnit(pu, force)

    if self.gravityKnockout then
        pcall(function()
            if pu.motion ~= nil then
                pu.motion:Knockout()
            end
        end)
    end

    -- Prova anche sul root, ma solo dopo i filtri anti-player.
    pcall(function()
        if pu.GetRootPu ~= nil then
            local root = pu:GetRootPu()

            if root ~= nil and root ~= pu and self:IsValidGravityTarget(root) then
                root:DisableGravityAWhile(self.gravityOffTime)
                root:SetKinematic(false)
                self:ApplyForceToPhysicsUnit(root, force * 0.95)

                if self.gravityKnockout and root.motion ~= nil then
                    root.motion:Knockout()
                end
            end
        end
    end)
end

function spellFireBall:ApplySlamForceToPhysicsUnit(pu, force)
    if pu == nil then return end
    if force == nil then return end

    -- Slam controllato: usiamo AddForce, ma NON il velocity assist aggressivo della fase di attrazione.
    pcall(function()
        pu:AddForce(force)
    end)

    pcall(function()
        if pu.rb ~= nil then
            pu.rb:AddForce(force)
        end
    end)

    -- Evita il lancio verso l'alto: se il ragdoll arriva allo slam con velocity.y positiva,
    -- la tagliamo verso il basso. Non tocchiamo x/z per non creare teletrasporti strani.
    pcall(function()
        if pu.rb ~= nil then
            local v = pu.rb.velocity
            if v ~= nil and v.y > self.gravityFinalVelocityClampY then
                pu.rb.velocity = UE.Vector3(v.x, self.gravityFinalVelocityClampY, v.z)
            end
        end
    end)

    -- Riduce lo spin in aria vicino al centro. Se angularVelocity non esiste, il pcall ignora.
    pcall(function()
        if pu.rb ~= nil and pu.rb.angularVelocity ~= nil then
            pu.rb.angularVelocity = pu.rb.angularVelocity * 0.35
        end
    end)
end

function spellFireBall:ApplySpikePushToPhysicsUnit(pu, center)
    if pu == nil then return end
    if center == nil then return end
    if not self:IsValidGravityTarget(pu) then return end

    local puPos = self:GetPhysicsUnitPosition(pu)
    if puPos == nil then return end

    local toCenter = center - puPos
    toCenter.y = 0

    local centerDir = UE.Vector3(0, 0, 0)

    if toCenter.sqrMagnitude > 0.001 then
        centerDir = toCenter.normalized
    end

    -- Quando escono le spike, niente esplosione outward: compressione verso centro + giù.
    local force = centerDir * self.gravitySpikeCenterForce + UE.Vector3.down * self.gravitySpikeSlamForce

    pcall(function()
        pu:ClearDisableGravity()
    end)

    self:ApplySlamForceToPhysicsUnit(pu, force)

    if self.gravityKnockout then
        pcall(function()
            if pu.motion ~= nil then
                pu.motion:Knockout()
            end
        end)
    end

    -- Sul root liberiamo la gravità/knockout, ma NON diamo un secondo impulso.
    -- Questo evita lo spappolamento e i lanci casuali del ragdoll.
    pcall(function()
        if pu.GetRootPu ~= nil then
            local root = pu:GetRootPu()

            if root ~= nil and root ~= pu and self:IsValidGravityTarget(root) then
                root:ClearDisableGravity()

                if self.gravityKnockout and root.motion ~= nil then
                    root.motion:Knockout()
                end
            end
        end
    end)
end

        toCenter.y = 0

        if toCenter.sqrMagnitude > 0.001 then
            -- Taglio/pressione verso centro e basso, per evitare di spararli in cielo.

        end

    return dir

function spellFireBall:ApplySlashDamageToPhysicsUnit(pu, center)

    end)

function spellFireBall:ApplyKurohitsugiSlashDamagePulse()
    if not self.kurohitsugiSlashDamageEnabled then return end

    local center = self:GetKurohitsugiGravityCenter()
    local targets = nil

    -- Prima proviamo i logic target: più sicuro e meno caotico.
    pcall(function()
        targets = CL.PhysicsUnit.QueryLogicTargetInRadiusTemp(center, self.gravityZoneRadius)
    end)

    local count = self:GetListCount(targets)

    for i = 0, count - 1 do
        local pu = self:GetListItem(targets, i)
        self:ApplySlashDamageToPhysicsUnit(pu, center)
    end

    -- Poi, se serve, proviamo anche le parti fisiche: qui NON applichiamo forza, solo hit scan damage.
    if self.kurohitsugiSlashUsePhysicsUnits then
        local allTargets = nil

        pcall(function()
            allTargets = CL.PhysicsUnit.QueryInRadiusTemp(center, self.gravityZoneRadius)
        end)

        local allCount = self:GetListCount(allTargets)

        for i = 0, allCount - 1 do
            local pu = self:GetListItem(allTargets, i)
            self:ApplySlashDamageToPhysicsUnit(pu, center)
        end
    end

    print("KUROHITSUGI TEST - Slash damage pulse")
end

function spellFireBall:UpdateKurohitsugiSlashDamage()
    if not self.kurohitsugiSlashDamageEnabled then return end

    -- I danni extra partono con le spike e durano fino a poco prima della frantumazione.
    if self.sequenceTimer < self.spikesTime then return end
    if self.sequenceTimer >= self.breakPiecesTime then return end

    if self.sequenceTimer - self.kurohitsugiSlashLastPulseTime >= self.kurohitsugiSlashPulseInterval then
        self.kurohitsugiSlashLastPulseTime = self.sequenceTimer
        self:ApplyKurohitsugiSlashDamagePulse()
    end
end

function spellFireBall:IsTargetInsideSpikeShockWaveRing(pu, center)
    if pu == nil then return false end
    if center == nil then return false end

    local pos = self:GetPhysicsUnitPosition(pu)
    if pos == nil then return false end

    local offset = pos - center
    offset.y = 0

    local distance = offset.magnitude
    local inner = self.spikeGroundShockWaveInnerRadius or 2.0
    local outer = self.spikeGroundShockWaveOuterRadius or self.spikeGroundShockWaveRadius or 10.0

    if distance < inner then
        print("KUROHITSUGI TEST - DIRECT SHOCKWAVE skipped inside ring target="
            .. tostring(self:GetSafeName(pu))
            .. " distance=" .. tostring(distance))
        return false
    end

    if distance > outer then
        print("KUROHITSUGI TEST - DIRECT SHOCKWAVE skipped outside ring target="
            .. tostring(self:GetSafeName(pu))
            .. " distance=" .. tostring(distance))
        return false
    end

    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE ring target="
        .. tostring(self:GetSafeName(pu))
        .. " distance=" .. tostring(distance))

    return true
end

function spellFireBall:IsSpikeShockWaveValidTarget(pu)
    if pu == nil then return false end

    -- Evita la gemma / host dello script.
    if self.host ~= nil and pu == self.host then return false end
    if self.interact ~= nil and pu == self.interact then return false end

    -- Evita player e mani tramite unitType, non tramite nome "controller":
    -- molti nemici validi arrivano come "goblinsmall... controller".
    local isPlayer = false
    pcall(function()
        if pu.unitType == CL.Tagger.Tag.PlayerRole or pu.unitType == CL.Tagger.Tag.PlayerHand then
            isPlayer = true
        end
    end)
    if isPlayer then return false end

    local tr = self:GetPhysicsUnitTransform(pu)
    if self:IsTransformLinkedToPlayer(tr) then
        return false
    end

    -- Target ideale: role/nemico.
    local isRole = false
    pcall(function()
        isRole = pu.unitType == CL.Tagger.Tag.InteractRole
    end)
    if isRole then return true end

    -- Accetta anche controller/root di nemici che espongono questi componenti,
    -- anche se il nome contiene "controller".
    if pu.controller ~= nil or pu.muscleState ~= nil or pu.actor ~= nil or pu.motion ~= nil then
        return true
    end

    -- Ultimo filtro leggero: esclude solo oggetti chiaramente nostri/preview/mani.
    local n = self:GetSafeName(pu)
    if n ~= nil and n ~= "" then
        local blocked = {
            "player",
            "hand",
            "glove",
            "mountpoint",
            "palmpreview",
            "targetarea",
            "preview",
            "exp_kurohitsugi_gem",
            "kurohitsugi_rainstring"
        }

        for i = 1, #blocked do
            if string.find(n, blocked[i]) ~= nil then
                return false
            end
        end
    end

    return false
end

function spellFireBall:GetSpikeShockWaveRoleTarget(pu)
    if pu == nil then return nil end

    -- Se è già un role target, usalo direttamente.
    local isRole = false
    pcall(function()
        isRole = pu.unitType == CL.Tagger.Tag.InteractRole
    end)

    if isRole then
        return pu
    end

    -- Alcune query fisiche tornano ossa/parti. Proviamo a risalire al root.
    local root = nil
    pcall(function()
        if pu.GetRootPu ~= nil then
            root = pu:GetRootPu()
        end
    end)

    if root ~= nil then
        local rootIsRole = false
        pcall(function()
            rootIsRole = root.unitType == CL.Tagger.Tag.InteractRole
        end)
        if rootIsRole then
            return root
        end

        -- Se il root sembra un controller/role nemico, usiamo lui come target principale.
        if root.controller ~= nil or root.muscleState ~= nil or root.actor ~= nil or root.motion ~= nil then
            return root
        end
    end

    return pu
end

function spellFireBall:ApplySpikeGroundShockWaveToPhysicsUnit(pu, center)
    if pu == nil then return false end
    if center == nil then return false end

    local target = self:GetSpikeShockWaveRoleTarget(pu)
    if target == nil then target = pu end

    if not self:IsTargetInsideSpikeShockWaveRing(target, center) then
        return false
    end

    if not self:IsSpikeShockWaveValidTarget(target) then
        print("KUROHITSUGI TEST - DIRECT SHOCKWAVE skipped target=" .. tostring(self:GetSafeName(target)))
        return false
    end

    local col = self:GetPhysicsUnitCollider(target)
    if col == nil then
        col = self:GetPhysicsUnitCollider(pu)
    end

    local dir = self:GetSpikeShockWaveDir(target, center)
    local appliedSomething = false

    -- 1) Danno stile Stomp/FlySlash: BeginScanDmg -> ApplyHitScanDamage -> EndHitScan.
    if self.spikeGroundShockWaveDamageEnabled and col ~= nil then
        local host = self:GetKurohitsugiDamageHost()

        if host ~= nil then
            local okDmg = pcall(function()
                CL.InteractTrigger.BeginScanDmg()
                CL.InteractTrigger.ApplyHitScanDamage(col, host, host.rb, host.col, dir, host.camp)
                CL.InteractTrigger.EndHitScan()
            end)

            if okDmg then
                appliedSomething = true
                print("KUROHITSUGI TEST - DIRECT SHOCKWAVE damage applied")
            end
        else
            print("KUROHITSUGI WARNING - DIRECT SHOCKWAVE damage host nil")
        end
    end

    -- 2) Reazione muscolare stile Stomp: stiff + clear stab + dizzy.
    pcall(function()
        if target.muscleState ~= nil then
            target.muscleState:DoStiff(self.spikeGroundShockWaveDoStiffTime or 1.0)
            target.muscleState:ClearAllStab()
            target.muscleState:EnterDizzy(self.spikeGroundShockWaveDizzyTime or 0.6)
            appliedSomething = true
            print("KUROHITSUGI TEST - DIRECT SHOCKWAVE muscle reaction applied")
        end
    end)

    -- 3) Interrompe l'azione corrente, come nello Stomp.
    pcall(function()
        if target.actor ~= nil then
            target.actor:StopAction(false)
            appliedSomething = true
        end
    end)

    pcall(function()
        if target.motion ~= nil then
            target.motion:ProtectMotionFunctionForAWhile(self.spikeGroundShockWaveProtectTime or 2)
            appliedSomething = true
        end
    end)

    -- 4) Pushback vero sul controller, stesso approccio dello Stomp.
    pcall(function()
        if target.controller ~= nil then
            target.controller:AddPushback(center, self.spikeGroundShockWavePushbackDis or 10, true)
            appliedSomething = true
            print("KUROHITSUGI TEST - DIRECT SHOCKWAVE pushback applied")
        end
    end)

    -- 5) Reazione fisica aggressiva controllata: serve a capire se stiamo toccando il corpo giusto.
    local impulse = dir * (self.spikeGroundShockWaveOutwardForce or 95)
        + UE.Vector3.up * (self.spikeGroundShockWaveUpForce or 65)
        + UE.Vector3.down * (self.spikeGroundShockWaveDownForce or 0)

    pcall(function()
        if target.DisableGravityAWhile ~= nil then
            target:DisableGravityAWhile(self.spikeGroundShockWaveGravityOffTime or 0.35)
        end
    end)

    pcall(function()
        if target.SetKinematic ~= nil then
            target:SetKinematic(false)
        end
    end)

    pcall(function()
        if target.AddForce ~= nil then
            target:AddForce(impulse, UE.ForceMode.Impulse)
            appliedSomething = true
            print("KUROHITSUGI TEST - DIRECT SHOCKWAVE target AddForce applied")
        end
    end)

    pcall(function()
        if pu ~= target and pu.AddForce ~= nil then
            pu:AddForce(impulse, UE.ForceMode.Impulse)
            appliedSomething = true
            print("KUROHITSUGI TEST - DIRECT SHOCKWAVE pu AddForce applied")
        end
    end)

    pcall(function()
        local rb = self:GetPhysicsUnitRigidbody(target)
        if rb == nil then
            rb = self:GetPhysicsUnitRigidbody(pu)
        end

        if rb ~= nil then
            rb.isKinematic = false
            rb:AddForce(impulse, UE.ForceMode.Impulse)
            rb:AddTorque(UE.Random.insideUnitSphere * 50, UE.ForceMode.Impulse)
            appliedSomething = true
            print("KUROHITSUGI TEST - DIRECT SHOCKWAVE rb impulse applied")
        end
    end)

    -- 6) Knockout/throwing opzionale per test visivo. Se funziona, poi lo abbassiamo.
    if self.spikeGroundShockWaveUseKnockout then
        pcall(function()
            if target.motion ~= nil then
                target.motion:Knockout()
                appliedSomething = true
                print("KUROHITSUGI TEST - DIRECT SHOCKWAVE knockout applied")
            end
        end)
    end

    if self.spikeGroundShockWaveUseThrowing then
        pcall(function()
            if target.muscleState ~= nil then
                target.muscleState:DoThrowing()
                appliedSomething = true
                print("KUROHITSUGI TEST - DIRECT SHOCKWAVE throwing applied")
            end
        end)
    end

    return appliedSomething
end

function spellFireBall:SpawnSpikeGroundShockWave()
    -- Nome mantenuto per compatibilità col punto di chiamata già esistente.
    -- Ora NON spawna più un FlyObj: applica direttamente l'effetto fisico Stomp-like.
    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE CALLED")

    if not self.spikeGroundShockWaveEnabled then
        print("KUROHITSUGI WARNING - DIRECT SHOCKWAVE disabled")
        return
    end

    if self.spikeGroundShockWaveSpawned then
        print("KUROHITSUGI WARNING - DIRECT SHOCKWAVE already used")
        return
    end

    self.spikeGroundShockWaveSpawned = true

    local center = self:GetSpikeGroundShockWaveCenter()
    local radius = self.spikeGroundShockWaveOuterRadius or self.spikeGroundShockWaveRadius or self.gravityZoneRadius or 10.0
    local inner = self.spikeGroundShockWaveInnerRadius or 2.0

    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE center=" .. tostring(center)
        .. " inner=" .. tostring(inner)
        .. " outer=" .. tostring(radius))

    local logicTargets = nil
    local physicsTargets = nil

    local okLogic = pcall(function()
        logicTargets = CL.PhysicsUnit.QueryLogicTargetInRadiusTemp(center, radius)
    end)

    local logicCount = self:GetListCount(logicTargets)
    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE logic ok=" .. tostring(okLogic) .. " count=" .. tostring(logicCount))

    local okPhysics = pcall(function()
        physicsTargets = CL.PhysicsUnit.QueryInRadiusTemp(center, radius)
    end)

    local physicsCount = self:GetListCount(physicsTargets)
    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE physics ok=" .. tostring(okPhysics) .. " count=" .. tostring(physicsCount))

    local appliedCount = 0
    local appliedMap = {}

    for i = 0, logicCount - 1 do
        local pu = self:GetListItem(logicTargets, i)
        local target = self:GetSpikeShockWaveRoleTarget(pu) or pu

        if target ~= nil and not appliedMap[target] then
            appliedMap[target] = true
            if self:ApplySpikeGroundShockWaveToPhysicsUnit(target, center) then
                appliedCount = appliedCount + 1
            end
        end
    end

    -- Fallback fisico sempre attivo, ma deduplicato sul root/role target.
    for i = 0, physicsCount - 1 do
        local pu = self:GetListItem(physicsTargets, i)
        local target = self:GetSpikeShockWaveRoleTarget(pu) or pu

        if target ~= nil and not appliedMap[target] then
            appliedMap[target] = true
            if self:ApplySpikeGroundShockWaveToPhysicsUnit(target, center) then
                appliedCount = appliedCount + 1
            end
        end
    end

    print("KUROHITSUGI TEST - DIRECT SHOCKWAVE applied=" .. tostring(appliedCount))
end

function spellFireBall:ApplyGravityFinalSlamToPhysicsUnit(pu, center)
    if pu == nil then return end
    if center == nil then return end
    if not self:IsValidGravityTarget(pu) then return end

    local puPos = self:GetPhysicsUnitPosition(pu)
    if puPos == nil then return end

    local toCenter = center - puPos
    toCenter.y = 0

    local centerDir = UE.Vector3(0, 0, 0)

    if toCenter.sqrMagnitude > 0.001 then
        centerDir = toCenter.normalized
    end

    -- Chiusura gravity prima della frantumazione: soprattutto verso il basso.
    local slamForce = UE.Vector3.down * self.gravityFinalSlamForce + centerDir * self.gravityFinalCenterForce

    pcall(function()
        pu:ClearDisableGravity()
    end)

    self:ApplySlamForceToPhysicsUnit(pu, slamForce)

    if self.gravityKnockout then
        pcall(function()
            if pu.motion ~= nil then
                pu.motion:Knockout()
            end
        end)
    end

    pcall(function()
        if pu.GetRootPu ~= nil then
            local root = pu:GetRootPu()

            if root ~= nil and root ~= pu and self:IsValidGravityTarget(root) then
                root:ClearDisableGravity()
                if self.gravityKnockout and root.motion ~= nil then
                    root.motion:Knockout()
                end
            end
        end
    end)
end

function spellFireBall:ApplyGravityZoneFinalSlam()
    if self.gravityFinalSlamDone then return end

    self.gravityFinalSlamDone = true

    local center = self:GetKurohitsugiGravityCenter()
    local targets = nil

    -- Stop finale solo sui logic target. Non QueryInRadiusTemp qui: quella era la causa più probabile
    -- di spin, doppie forze e lanci casuali verso l'alto.
    pcall(function()
        targets = CL.PhysicsUnit.QueryLogicTargetInRadiusTemp(center, self.gravityZoneRadius)
    end)

    local count = self:GetListCount(targets)

    for i = 0, count - 1 do
        local pu = self:GetListItem(targets, i)
        self:ApplyGravityFinalSlamToPhysicsUnit(pu, center)
    end

    print("KUROHITSUGI TEST - Gravity final controlled stop")
end

function spellFireBall:ApplyGravityZoneTick()
    local center = self:GetKurohitsugiGravityCenter()

    local targets = nil

    pcall(function()
        targets = CL.PhysicsUnit.QueryLogicTargetInRadiusTemp(center, self.gravityZoneRadius)
    end)

    local count = self:GetListCount(targets)

    for i = 0, count - 1 do
        local pu = self:GetListItem(targets, i)
        self:ApplyGravityToPhysicsUnit(pu, center)
    end

    -- Opzionale futuro: attrarre anche oggetti fisici/materia.
    -- Per ora è spento per evitare di tirare dentro ambiente/oggetti inutili.
    if self.gravityAffectObjects then
        local allTargets = nil

        pcall(function()
            allTargets = CL.PhysicsUnit.QueryInRadiusTemp(center, self.gravityZoneRadius)
        end)

        local allCount = self:GetListCount(allTargets)

        for i = 0, allCount - 1 do
            local pu = self:GetListItem(allTargets, i)
            self:ApplyGravityToPhysicsUnit(pu, center)
        end
    end
end

function spellFireBall:GetGravityStartTime()
    -- Parte con un piccolo delay dopo l'inizio delle pareti.
    -- Così la gravità entra quando il Kurohitsugi sta iniziando a chiudersi,
    -- senza aspettare le spike e senza partire troppo presto sulle sole linee.
    return self.wallsStartTime + self.gravityStartAfterWallsStart
end


function spellFireBall:GetInnerDarknessStartTime()
    -- L'oscurità interna deve comparire solo dopo che il top ha finito di chiudersi.
    -- Usiamo anche topDelay per evitare che appaia durante l'animazione visibile del top.
    return self.topStartTime + self.topGrowDuration + self.topDelay + self.innerDarknessStartAfterTopComplete
end

function spellFireBall:StartKurohitsugiGravityZone()
    if self.gravityZoneStarted then return end

    self.gravityZoneStarted = true

    -- Termina poco prima della frantumazione, così i nemici non restano sospesi troppo.
    self.gravityZoneEndTime = self.breakPiecesTime - self.gravityEndBeforeBreak

    self.gravityZoneLastTick = -999

    -- Primo tick immediato quando parte davvero la zona.
    self:ApplyGravityZoneTick()

    print("KUROHITSUGI TEST - Gravity zone started")
end

function spellFireBall:UpdateKurohitsugiGravityZone()
    if not self.gravityZoneStarted then return end

    if self.sequenceTimer >= self.gravityZoneEndTime then
        self:ApplyGravityZoneFinalSlam()
        self.gravityZoneStarted = false
        return
    end

    if self.sequenceTimer - self.gravityZoneLastTick >= self.gravityZoneTickInterval then
        self.gravityZoneLastTick = self.sequenceTimer
        self:ApplyGravityZoneTick()
    end
end

function spellFireBall:IsRainFlyObjectConfigured()
    if not self.kurohitsugiFlyRainEnabled then return false end
    if self.kurohitsugiRainFlyObject == nil then return false end
    if self.kurohitsugiRainFlyObject == "" then return false end
    if self.kurohitsugiRainFlyObject == "NOME_FLYOBJECT_RAPIER_O_SLASH" then return false end
    return true
end

function spellFireBall:GetRainOffset(index, pulseIndex)
    local count = self.kurohitsugiRainObjectsPerPulse
    if count == nil or count < 1 then count = 1 end

    -- Distribuzione più larga e meno "a imbuto" verso il centro.
    -- Con 6 dardi: 4/5 restano in anello esterno, 1/2 su anello medio.
    local angle = ((index - 1) / count) * 6.28318 + pulseIndex * 0.48
    local ring01 = 1.00

    if ((index + pulseIndex) % 3) == 0 then
        ring01 = 0.58
    elseif ((index + pulseIndex) % 3) == 1 then
        ring01 = 0.82
    end

    local r = self.kurohitsugiRainRadius * ring01

    return UE.Vector3(UE.Mathf.Cos(angle) * r, 0, UE.Mathf.Sin(angle) * r)
end

function spellFireBall:SpawnKurohitsugiRainFlyObject(pos, dir)
    if not self:IsRainFlyObjectConfigured() then return end
    if pos == nil or dir == nil then return end

    -- CreateFlyObj è in pcall: se il nome FlyObject è sbagliato, non rompiamo la spell.
    pcall(function()
        CL.FlyObject.CreateFlyObj(self.kurohitsugiRainFlyObject, self.interact, pos, dir)
    end)
end

function spellFireBall:SpawnKurohitsugiFlyRainPulse()
    if not self:IsRainFlyObjectConfigured() then return end
    if self.castRoot == nil then return end

    if self.kurohitsugiRainMaxPulseCount ~= nil and self.kurohitsugiRainMaxPulseCount > 0 then
        if self.kurohitsugiRainPulseIndex >= self.kurohitsugiRainMaxPulseCount then
            return
        end
    end

    self.kurohitsugiRainPulseIndex = self.kurohitsugiRainPulseIndex + 1

    local base = self.castRoot.position
    local count = self.kurohitsugiRainObjectsPerPulse
    if count == nil or count < 1 then count = 1 end

    local inward = self.kurohitsugiRainTargetInwardMultiplier
    if inward == nil then inward = 0.82 end

    for i = 1, count do
        local offset = self:GetRainOffset(i, self.kurohitsugiRainPulseIndex)

        local spawnPos = base + offset
        spawnPos.y = base.y + self.kurohitsugiRainSpawnHeight

        -- Prima puntava molto al centro: i dardi si concentravano tutti nello stesso punto.
        -- Ora restano quasi verticali e distribuiti dentro il cubo.
        local targetPos = base + offset * inward
        targetPos.y = base.y + self.kurohitsugiRainTargetHeight

        local dir = targetPos - spawnPos
        if dir.sqrMagnitude < 0.001 then
            dir = UE.Vector3.down
        else
            dir = dir.normalized
        end

        self:SpawnKurohitsugiRainFlyObject(spawnPos, dir)
    end

    print("KUROHITSUGI TEST - FlyObject rain pulse")
end

function spellFireBall:UpdateKurohitsugiFlyRain()
    if not self:IsRainFlyObjectConfigured() then return end
    if not self.kurohitsugiRainUseAutoPulses then return end

    local startTime = self.spikesTime + self.kurohitsugiRainStartDelayFromSpikes
    local endTime = self.breakPiecesTime - self.kurohitsugiRainStopBeforeBreak

    if self.sequenceTimer < startTime then return end
    if self.sequenceTimer >= endTime then return end

    if self.kurohitsugiRainMaxPulseCount ~= nil and self.kurohitsugiRainMaxPulseCount > 0 then
        if self.kurohitsugiRainPulseIndex >= self.kurohitsugiRainMaxPulseCount then
            return
        end
    end

    if self.sequenceTimer - self.kurohitsugiRainLastPulseTime >= self.kurohitsugiRainPulseInterval then
        self.kurohitsugiRainLastPulseTime = self.sequenceTimer
        self:SpawnKurohitsugiFlyRainPulse()
    end
end

function spellFireBall:StartSequenceReiatsuContinuity()
    if self.startedSequenceReiatsu then return end

    self:FindKurohitsugiObjects()

    self.startedSequenceReiatsu = true
    self.stoppedSequenceReiatsu = false

    if self.sequenceReiatsuRoot ~= nil then
        SetActive(self.sequenceReiatsuRoot, true)
        PlayParticleSystemsInRoot(self.sequenceReiatsuRoot)
        print("KUROHITSUGI TEST - Sequence Reiatsu continuity started")
    else
        print("KUROHITSUGI WARNING - SequenceReiatsuRoot nil")
    end

    -- Continuità anche sugli effetti sui nemici:
    -- la Charge Reiatsu Bind non viene fermata al release, ma viene aggiornata
    -- durante la sequenza fino alla comparsa delle spike/glow.
    if self.chargeReiatsuActive then
        self.chargeReiatsuLastTick = -999
        self:ApplyChargeReiatsuTick()
    end
end

function spellFireBall:StopSequenceReiatsuContinuity()
    if self.stoppedSequenceReiatsu then return end

    self.stoppedSequenceReiatsu = true

    StopParticleSystemsInRoot(self.sequenceReiatsuRoot)
    SetActive(self.sequenceReiatsuRoot, false)

    self:StopChargeReiatsuBind()

    print("KUROHITSUGI TEST - Sequence Reiatsu continuity stopped")
end

function spellFireBall:UpdateSequenceReiatsuContinuity()
    if not self.startedSequenceReiatsu then return end
    if self.stoppedSequenceReiatsu then return end

    -- La continuità dura dall'inizio della sequenza fino alla comparsa delle spike/glow.
    -- Così la pressione spirituale accompagna linee, pareti, top e prima chiusura,
    -- poi lascia spazio alla fase offensiva vera: glow, seconda rain, spike e frantumazione.
    if self.sequenceTimer >= self.spikesTime then
        self:StopSequenceReiatsuContinuity()
        return
    end

    self:UpdateChargeReiatsuBind()
end


function spellFireBall:StartGlowTransition()
    if self.glowTransitionRoot == nil then return end
    if self.startedGlowTransition then return end

    self.startedGlowTransition = true
    self.stoppedGlowTransition = false

    SetActive(self.glowTransitionRoot, true)
    PlayParticleSystemsInRoot(self.glowTransitionRoot)

    print("KUROHITSUGI TEST - Glow transition started")
end

function spellFireBall:StopGlowTransition()
    if self.glowTransitionRoot == nil then return end
    if self.stoppedGlowTransition then return end

    self.stoppedGlowTransition = true

    StopParticleSystemsInRoot(self.glowTransitionRoot)
    SetActive(self.glowTransitionRoot, false)

    print("KUROHITSUGI TEST - Glow transition stopped")
end

function spellFireBall:UpdateGlowTransition()
    if self.glowTransitionRoot == nil then return end

    if self.sequenceTimer >= self.glowTransitionTime and not self.startedGlowTransition then
        self:StartGlowTransition()
    end

    if self.startedGlowTransition
        and not self.stoppedGlowTransition
        and self.sequenceTimer >= self.glowTransitionTime + self.glowTransitionDuration then

        self:StopGlowTransition()
    end
end

function spellFireBall:UpdateFirstRainPulseAfterCubeComplete()
    if self.startedFirstRainPulse then return end
    if not self:IsRainFlyObjectConfigured() then return end

    if self.sequenceTimer >= self.firstRainTime then
        self.startedFirstRainPulse = true
        self.kurohitsugiRainLastPulseTime = self.sequenceTimer
        self:SpawnKurohitsugiFlyRainPulse()
        print("KUROHITSUGI TEST - First rain pulse after cube complete")
    end
end

function spellFireBall:UpdateSecondRainPulseWithGlow()
    if self.startedSecondRainPulse then return end
    if not self:IsRainFlyObjectConfigured() then return end

    if self.sequenceTimer >= self.secondRainTime then
        self.startedSecondRainPulse = true
        self.kurohitsugiRainLastPulseTime = self.sequenceTimer
        self:SpawnKurohitsugiFlyRainPulse()
        print("KUROHITSUGI TEST - Second rain pulse with glow")
    end
end

function spellFireBall:UpdateKurohitsugiSequence()
    self.sequenceTimer = self.sequenceTimer + UE.Time.deltaTime

    self:LockCastTransform()

    self:UpdateSequenceReiatsuContinuity()

    self:AnimateLines()

    if self.sequenceTimer >= self.wallsStartTime then
        self:AnimateWalls()
    end

    if self.sequenceTimer >= self:GetInnerDarknessStartTime() and not self.startedInnerDarkness then
        self.startedInnerDarkness = true
        SetActive(self.innerDarknessRoot, true)
    end

    if self.sequenceTimer >= self.topStartTime then
        self:AnimateTop()
    end

    if self.sequenceTimer >= self.effectsTime and not self.startedEffects then
        self.startedEffects = true
        SetActive(self.effectsRoot, true)
    end

    -- La gravity zone parte dopo le linee + delay extra, così non si vede troppo presto prima dei muri.
    if self.sequenceTimer >= self:GetGravityStartTime() and not self.gravityZoneStarted and not self.gravityFinalSlamDone then
        self:StartKurohitsugiGravityZone()
    end

    self:UpdateKurohitsugiGravityZone()
    self:UpdateKurohitsugiSlashDamage()
    self:UpdateKurohitsugiFlyRain()
    self:UpdateFirstRainPulseAfterCubeComplete()
    self:UpdateGlowTransition()
    self:UpdateSecondRainPulseWithGlow()

    if self.sequenceTimer >= self.spikesTime and self.sequenceTimer < self.spikeRetractTime then
        if not self.startedSpikes then
            self.startedSpikes = true

            if self.spikesRoot ~= nil then
                self.spikesRoot.gameObject:SetActive(true)
            end

            -- Push controllato sincronizzato con l'uscita delle spike.
            -- La shockwave fisica viene spawnata dentro ApplyGravitySpikePush(),
            -- così parte nello stesso momento del push e non viene chiamata due volte.
            self:ApplyGravitySpikePush()

            -- Primo pulse immediato di danno interno stile slash.
            self.kurohitsugiSlashLastPulseTime = -999
            self:ApplyKurohitsugiSlashDamagePulse()

            -- La rain non parte più con le spike: ora è controllata da
            -- UpdateFirstRainPulseAfterCubeComplete() e UpdateSecondRainPulseWithGlow().
        end

        self:AnimateSpikesGrow()
    end

    if self.sequenceTimer >= self.spikeRetractTime and self.sequenceTimer < self.breakPiecesTime then
        if not self.retractedSpikes then
            self.retractedSpikes = true

            if self.spikesRoot ~= nil then
                self.spikesRoot.gameObject:SetActive(true)
            end
        end

        self:AnimateSpikesRetract()
    end

    if self.sequenceTimer >= self.breakPiecesTime and self.sequenceTimer < self.hideTime then
        if not self.startedBreakPieces then
            self.startedBreakPieces = true

            -- Slam sincronizzato con la frantumazione: se la zona non è già finita, lo forziamo qui.
            self:ApplyGravityZoneFinalSlam()

            SetActive(self.linesRoot, false)
            SetActive(self.wallsRoot, false)
            SetActive(self.topRoot, false)
            SetActive(self.innerDarknessRoot, false)
            self:StopGlowTransition()

            if self.spikesRoot ~= nil then
                SetChildrenActive(self.spikesRoot, false)
                self.spikesRoot.gameObject:SetActive(false)
            end

            SetActive(self.breakPiecesRoot, true)
        end

        self:AnimateBreakPieces()
    end

    if self.sequenceTimer >= self.hideTime then
        self:HideKurohitsugiCastAll()
    end
end

function spellFireBall:StartKurohitsugiSequence()
    if self.isSequenceRunning then
        print("KUROHITSUGI BLOCKED - sequence already running")
        return
    end

    self.isSequenceRunning = true
    self.pendingCastBlocked = false

    self:StopKurohitsugiSequence()
    self:SetKurohitsugiInitialVisualState()
    self:StartSequenceReiatsuContinuity()

    local owner = self

    self.sequenceSchedId = CL.Scheduler.Create({},
        function()
            if owner ~= nil then
                owner:UpdateKurohitsugiSequence()
            end
        end,
    0, -1).actionId
end

function spellFireBall:ResetKurohitsugiCast()
    self:FindKurohitsugiObjects()
    self:StopKurohitsugiSequence()

    self.castLockedPos = nil
    self.castLockedRot = nil

    self.gravityZoneStarted = false
    self.gravityZoneEndTime = 0
    self.gravityZoneLastTick = 0
    self.gravitySpikePushDone = false
    self.gravityFinalSlamDone = false

    self:StopChargeReiatsuBind()

    self.kurohitsugiSlashLastPulseTime = -999
    self.kurohitsugiSlashStarted = false
    self.kurohitsugiRainLastPulseTime = -999
    self.kurohitsugiRainPulseIndex = 0
    self.startedFirstRainPulse = false
    self.startedSecondRainPulse = false
    self.startedGlowTransition = false
    self.stoppedGlowTransition = false

    if self.castRoot ~= nil then
        self.castRoot:SetParent(self.trans, true)
        self.castRoot.gameObject:SetActive(false)

        self.castRoot.localPosition = UE.Vector3.zero
        self.castRoot.localRotation = UE.Quaternion.identity
    end

    self:SetKurohitsugiInitialVisualState()
end

function spellFireBall:CastKurohitsugiAtTarget()
    if self.isSequenceRunning then
        self.pendingCastBlocked = true
        print("KUROHITSUGI BLOCKED - active sequence, cast ignored")
        return
    end

    if self.lastTargetPos == nil then
        print("KUROHITSUGI ERROR - lastTargetPos nil")
        return
    end

    self:FindKurohitsugiObjects()

    if self.castRoot == nil then
        print("KUROHITSUGI ERROR - KurohitsugiCastRoot nil")
        return
    end

    local pos = UE.Vector3(self.lastTargetPos.x, self.lastTargetPos.y, self.lastTargetPos.z)
    pos = self:GetGroundedPosition(pos, self.castHeightOffset)

    self.castRoot:SetParent(nil, true)

    self.castRoot.position = pos
    self.castRoot.rotation = UE.Quaternion.identity
    self.castRoot.gameObject:SetActive(true)

    self.castLockedPos = UE.Vector3(pos.x, pos.y, pos.z)
    self.castLockedRot = UE.Quaternion.identity

    self:StartKurohitsugiSequence()

    print("KUROHITSUGI TEST - Final animated sequence started")
end

function spellFireBall:OnChargeBegin()
    -- Se il Kurohitsugi è già attivo, non apriamo nemmeno il preview.
    -- Questo impedisce che un nuovo cast resetti la sequenza precedente.
    if self.isSequenceRunning then
        self.pendingCastBlocked = true
        print("KUROHITSUGI BLOCKED - charge ignored while sequence is running")
        return
    end

    -- Mantiene la mano in posa spell aperta durante il charge,
    -- come fa la Fireball base in WMD_SpellThrowScript.
    pcall(function()
        self:SetHandPose(true)
    end)

    self:ResetKurohitsugiCast()

    self:ShowKurohitsugiPreview()
    self:InitTargetPosition()
    self:ShowReiatsuVisual()
    self:StartKurohitsugiPreviewUpdate()
    self:StartChargeReiatsuBind()
end

function spellFireBall:OnChargeRelease()
    -- Riporta la mano alla posa normale quando finisce il charge.
    pcall(function()
        self:SetHandPose(false)
    end)

    -- Non fermiamo subito la Reiatsu Bind:
    -- deve continuare nei primi istanti della sequenza, insieme al SequenceReiatsuRoot.
    self:HideReiatsuVisual()

    -- Se durante una sequenza attiva arriva un release, lo ignoriamo.
    -- Importante: non chiamiamo ResetKurohitsugiCast e non fermiamo la sequenza in corso.
    if self.isSequenceRunning then
        self.pendingCastBlocked = true
        self:HideKurohitsugiPreviewAll()
        self:StopChargeReiatsuBind()
        print("KUROHITSUGI BLOCKED - release ignored while sequence is running")
        return
    end

    local castPos = self.lastTargetPos

    self:HideKurohitsugiPreviewAll()

    if castPos ~= nil then
        self.lastTargetPos = castPos
        self:CastKurohitsugiAtTarget()
    else
        self:StopChargeReiatsuBind()
    end
end

function spellFireBall:OnChargeCancel()
    -- Riporta la mano alla posa normale anche se il charge viene cancellato.
    pcall(function()
        self:SetHandPose(false)
    end)

    self:StopChargeReiatsuBind()
    self:HideReiatsuVisual()

    if self.isSequenceRunning then
        return
    end

    self:HideKurohitsugiPreviewAll()
end
